**思想**：

* 字符串模拟。
  * 对于第一个符合要求的年份，可以暴力枚举日期进行判断。
  * 第二个年份，需要保证 `ABABBABA` 的形式，则只需枚举 `AB` 即可。

**代码**：
```cpp
#include&lt;bits/stdc++.h&gt;
using namespace std;

int y[2][13] = {{0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31},
                {0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31}};

int is_r(int x){  //判断是否是闰年
    if((x % 4 == 0 && x % 100 != 0) || (x % 100 == 0 && x % 400 == 0)) return 1;
    else return 0;
}

int Stoi(string s){  //蓝桥杯不支持将string转换为int类型的函数，手搓一个
    int sum = 0;
    for(int i = 0; i &lt; s.size(); i ++){
        sum = sum * 10 + int(s[i] - '0');
    }
    return sum;
}

bool is_t(string x, int p){  //判断该年份是否合法，即月份和天数是否合法
    int m = Stoi(x.substr(0, 2));
    int d = Stoi(x.substr(2, 2));
    if(m &gt;= 1 && m &lt;= 12 && d &lt;= y[is_r(p)][m]) return true;
    return false;
}

string To_string(int num){  //蓝桥杯不支持将int转换为string类型的函数，手搓一个
    if(num == 0) return "0";
    string sum = "";
    while(num){
        int t = num % 10;
        sum += char(t + '0');
        num /= 10;
    }
    reverse(sum.begin(), sum.end());
    return sum;
}

bool is_b(string a, string b){  //判断这一年是否在所给年份之后
    int m1 = Stoi(a.substr(0, 2)), m2 = Stoi(b.substr(0, 2));
    if(m1 == m2){
        int d1 = Stoi(a.substr(2, 2)), d2 = Stoi(b.substr(2, 2));
        return d1 &gt; d2;
    }
    else return m1 > m2;
}
```cpp
string to_y(int a, int b, int c){  //将年月日拼接为YYYYMMDD形式
    string sum = "";
    string A = to_string(a);
    while(A.size() &lt; 4) A = "0" + A;
    string B = to_string(b);
    while(B.size() &lt; 2) B = "0" + B;
    string C = to_string(c);
    while(C.size() &lt; 2) C = "0" + C;
    return A + B + C;
}

void solve(){
    string s; cin &gt;> s;
    string t = s.substr(0, 4);  //取出年
    string k = s.substr(4, 4);  //取出月份和日期
    int year = stoi(t);
    int Y = year;
    int m = stoi(k.substr(0, 2));  //取出月份
    int d = stoi(k.substr(2, 2));  //取出日期

    while(1){  //暴力枚举
        d++;  //日期++
        if(d > y[is_r(Y)][m]){
            d = 1, m++;  //超过该月的最大日期，月份++
        }
        if(m > 12) Y++, m = 1;  //超过12月，年份++
        string s1 = to_y(Y, m, d);  //转化为YYYYMMDD的形式
        string s2 = s1; 
        reverse(s2.begin(), s2.end());
        if(s1 == s2){  //判断是否回文
            cout << s1 << endl;
            break;
        }
    }

    string ab = s.substr(0, 2);
    int yy = stoi(ab);  //ABAB形，先取出AB
    for(int i = 0; ; i++){
        int Y = yy + i;
        string a = to_string(Y);
        if(a[0] == a[1]) continue;  //AA直接跳过
        a += a;  //得到ABAB
        string aa = a;
        reverse(a.begin(), a.end());  //月份和日期必须为BABA
        int b = stoi(a);
        if(is_t(a, Y) && (is_b(a, k) || is_b(aa, t))){
            cout << aa << a << endl;
            break;
        }
    }
}

    solve();
    return 0;
}
