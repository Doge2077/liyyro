[原题链接](https://www.acwing.com/problem/content/3732/)

**思想**：

* 前缀和与差分。
  * 对于 `a[i]` 的操作，构建差分数组 `b[i]` 记录被操作的区间。
  * 利用前缀和可知所有被操作的区间，即 `b[i] != 0` 的区间。

**代码**：
```cpp
#include &lt;bits/stdc++.h&gt;
using namespace std;

const int N = 1e6 + 3;

int a[N], b[N];

void solve(){
    int n; cin >> n; 
    for(int i = 1; i &lt;= n; i ++){
        b[i] = 0;  //重置
        cin &gt;> a[i];  //记录操作
    }
    for(int i = 1; i &lt;= n; i ++){
        if(a[i] &gt;= i) b[1] ++, b[i + 1] --;  //a[i] 超出范围，则当前全部区间都被操作了
        else if(a[i] &lt; i && a[i] != 0) b[i - a[i] + 1] ++, b[i + 1] --;  //否则操作后 a[i] 个数，正数为第 i - a[i] + 1 的位置开始
    }
    for(int i = 1; i &lt;= n; i ++){
        b[i] += b[i - 1];  //前缀和计算每个区间被操作的次数
        if(b[i] != 0) cout &lt;&lt; 1 &lt;&lt; ' ';  //不为 0 说明该位置被操作过
        else cout &lt;&lt; 0 &lt;&lt; ' ';
    }
    cout &lt;&lt; endl;
}

int main(){
    int _; cin &gt;> _;
    while(_ --) solve();
    return 0;
}
```
